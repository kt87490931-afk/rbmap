<script>
(function () {
  var api = "https://rbbmap.com/api/track/click";
  var trackPath = "/p/threeno-swtest";
  var root = document.querySelector(".rb-slice");
  if (!root) return;
  root.addEventListener(
    "click",
    function (e) {
      var a = e.target && e.target.closest && e.target.closest("a[href^='tel:']");
      if (!a) return;
      fetch(api, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ type: "call", path: trackPath }),
        mode: "cors",
        keepalive: true,
      }).catch(function () {});
    },
    { passive: true }
  );
})();
</script>
